```jsx
for (let i = 0; i < jonas.length; i++) {
```

## Looping Backwards and Loops in Loops

```jsx
const jonas = [
	"Jonas",
	"Schmedtmann",
	2022 - 1991,
	"teacher",
	["Michael", "Peter", "Steven"],
];

for (let i = jonas.length - 1; i >= 0; i--) {
	console.log(jonas[i]);
}
```
